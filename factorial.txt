import java.util.Scanner;
class Factorial {
    public static void main(String[] args)
    {
        int num,i,fact=1;
        Scanner obj = new Scanner(System.in);
        System.out.println("enter number: ");
        num=obj.nextInt();
        for(i=1;i<=num;i++)
        {
            fact*=i;
        }
        System.out.println("factorial of "+num+ " is "+fact);
    }
}
