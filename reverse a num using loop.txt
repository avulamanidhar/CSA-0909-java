import java.util.Scanner;
public class Reversenum {
    public static void main(String[] args) {
        int num,rem;
        Scanner obj= new Scanner(System.in);
        System.out.println("enter num: ");
        num=obj.nextInt();
        System.out.print("reverse num is ");
        while(num>0)
        {
            rem=num%10;
            System.out.print(rem);
            num/=10;
        }
    }
}
